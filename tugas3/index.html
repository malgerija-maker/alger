🚗 Class Mobil dengan Named Constructor

Named constructor adalah constructor yang memiliki nama tambahan, sehingga dalam satu class dapat memiliki beberapa constructor dengan fungsi yang berbeda.

📋 NAMED CONSTRUCTOR - LENGKAP
// Nama   : M. Algerija hertyaz paslam
// Nim    : 251410021
// Kelas  : SI2A


import 'dart:convert';

class Mobil {
  String? merk;
  int? tahun;

  // Parameterized constructor
  Mobil(this.merk, this.tahun);

  // Named constructor
  Mobil.baru() {
    merk = 'Toyota';
    tahun = 2023;
  }

  // Named constructor dengan parameter
  Mobil.dariJson(Map json) {
    merk = json['merk'];
    tahun = json['tahun'];
  }

  // Override toString for better print output
  @override
  String toString() {
    return 'Mobil{merk: $merk, tahun: $tahun}';
  }
}

void main() {
  print('--- Demonstrasi Class Mobil ---');

  // 1. Menggunakan constructor default (parameterized constructor)
  var mobilA = Mobil('Honda', 2020);
  print('Mobil A (constructor default): $mobilA');

  // 2. Menggunakan named constructor Mobil.baru()
  var mobilB = Mobil.baru();
  print('Mobil B (named constructor .baru): $mobilB');

  // 3. Menggunakan named constructor Mobil.dariJson()
  String jsonString = '{"merk": "Suzuki", "tahun": 2021}';
  Map jsonData = jsonDecode(jsonString);
  var mobilC = Mobil.dariJson(jsonData);
  print('Mobil C (named constructor .dariJson): $mobilC');

  print('\n--- Akhir Demonstrasi ---');
}

🎨 Constant Constructor (Immutable)

Digunakan untuk objek yang tidak berubah. Semua field harus final. Menghemat memory untuk objek yang sama.

🎨 CLASS WARNA DENGAN CONST CONSTRUCTOR

class Warna {
  final int r, g, b;
  const Warna(this.r, this.g, this.b);

  @override
  String toString() {
    return 'Warna(r: $r, g: $g, b: $b)';
  }
}

void main() {
  const merah = Warna(255, 0, 0);
  const merahLagi = Warna(255, 0, 0); // Menggunakan instance yang sama di memory

  print(merah);
  print(merahLagi);
  print('Apakah identik? ${identical(merah, merahLagi)}');
}

🔒 Enkapsulasi & Private

Enkapsulasi adalah konsep dalam pemrograman berorientasi objek yang digunakan untuk melindungi data agar tidak dapat diakses secara langsung dari luar class. Dengan enkapsulasi, data hanya bisa diakses melalui method tertentu sehingga lebih aman dan terkontrol. Di Dart, enkapsulasi dilakukan dengan menambahkan underscore (_) di depan nama variabel untuk menjadikannya private.

👤 CLASS PERSON DENGAN PRIVATE FIELD
// Nama   : M. Algerija hertyaz paslam
// Nim    : 251410021
// Kelas  : SI2A

class Person {
  String _name; // Private: Hanya bisa diakses di file ini
  int age;      // Public: Dapat diakses bebas

  Person(this._name, this.age);

  void info() => print('Nama: $_name, Umur: $age');
}

void main() {
  var p = Person('M. Algerija hertyaz paslam', 18);
  p.info();
}

📤 Getter & Setter

Getter adalah method yang digunakan untuk mengambil nilai dari variabel private, sedangkan Setter adalah method yang digunakan untuk mengubah nilai variabel private dengan aturan tertentu.

🔍 Getter dan setter
// Nama   : M. Algerija hertyaz paslam
// Nim    : 251410021
// Kelas  : SI2A

class Person {
  String _name;
  int age;
  Person(this._name, this.age);
  
  // GETTER
  // 1. Menggunakan Arrow Syntax
  String get getName1 => _name;
  
  // 2. Menggunakan Block Syntax
  String get getName2 {
    return _name;
  }

  // SETTER
  // 1. Menggunakan Keyword 'set'
  set setName1(String name) {
    _name = name;
  }

  // 2. Menggunakan Method Approach
  void setNameManual(String name) {
    _name = name;
  }
}

void main() {
  Person p = Person("Alger", 18);
  print('Umur Awal: ${p.age}');
  
  // Test Setters
  p.setName1 = 'Paslam';
  p.setNameManual('M. Algerija hertyaz paslam');

  // Test Getters
  print('Hasil Getter 1: ${p.getName1}');
  print('Hasil Getter 2: ${p.getName2}');
}

class Produk {
  String _nama;
  double _harga;
  
  Produk(this._nama, this._harga);
  
  // Getter
  String get nama => _nama;
  double get harga => _harga;
  
  // Setter dengan validasi
  set nama(String value) {
    if (value.length >= 3) {
      _nama = value;
    } else {
      print('Gagal: Nama minimal 3 karakter.');
    }
  }
  
  set harga(double value) {
    if (value > 0) {
      _harga = value;
    } else {
      print('Gagal: Harga harus lebih dari 0.');
    }
  }
}

void main() {
  var produk1 = Produk('Monitor Samsung', 3500.0);
  print('Awal: ${produk1.nama} - Rp${produk1.harga}');
  
  // Update valid
  produk1.nama = 'Keyboard Mekanik';
  produk1.harga = 1200.0;
  print('Setelah update valid: ${produk1.nama} - Rp${produk1.harga}');
  
  // Update tidak valid
  produk1.nama = 'TV';      // gagal ( < 3 karakter)
  produk1.harga = -100.0;   // gagal ( <= 0)
  print('Setelah update gagal: ${produk1.nama} - Rp${produk1.harga}');
}

class User {

  User(this._username, this._password, this._email);

  // Getter untuk semua field
  String get username => _username;
  String get email => _email;

  set username(String v) {
    if (v.length >= 5) {
      _username = v;
    } else {
      print('Username min 5 karakter');
    }
  }

  set password(String v) {
    if (v.length >= 8) {
      _password = v;
    } else {
      print('Password min 8 karakter');
    }
  }

  set email(String v) {
    if (v.contains('@')) {
      _email = v;
    } else {
      print('Format email salah (Harus @)');
    }
  }

  void info() => print('User: $_username, Pass: $_password, Email: $_email');
}

class User {
  String _username;
  String _password;

  User(this._username, this._password);
  String get username => _username;

  bool login(String username, String password) {
    return _username == username && _password == password;
  }

  void gantiPassword(String lama, String baru) {
    if (_password == lama) {
      if (baru.length >= 8) {
        _password = baru;
        print('Password berhasil diubah.');
      } else {
        print('Gagal: Password baru minimal 8 karakter.');
      }
    } else {
      print('Gagal: Password lama tidak sesuai.');
    }
  }
}

void main() {
  print('--- Inisialisasi User ---');
  final User user1 = User('admin_Alger', 'securePass1611');
  
  print('--- Uji Coba Login ---');
  if (user1.login('admin_Alger', 'securePass1611')) {
    print('Login Berhasil!');
  }

  print('\n--- Uji Coba Ganti Password ---');
  user1.gantiPassword('securePass1611', 'newSecurePass2020');
  user1.gantiPassword('wrongPass', 'anyPass'); // Gagal
}

🏦 Latihan: RekeningBank

Buat kelas RekeningBank dengan Field private: _nomorRekening, _saldo, dan Method: deposit(), tarik(), cekSaldo() serta Validasi: saldo tidak bisa negatif.

💰 TEMPLATE REKENINGBANK
// Nama   : M. Algerija hertyaz paslam
// Nim    : 251410021
// Kelas  : SI2A

class RekeningBank {
  String _nomorRekening;
  double _saldo;

  RekeningBank(this._nomorRekening, this._saldo);

  void deposit(double jml) {
    if (jml > 0) {
      _saldo += jml;
      print('Deposit Rp $jml Berhasil.');
    }
  }

  void tarik(double jml) {
    if (jml > 0 && _saldo >= jml) {
      _saldo -= jml;
      print('Tarik Rp $jml Berhasil.');
    } else {
      print('Gagal: Saldo tidak cukup atau input tidak valid.');
    }
  }

  void cekSaldo() {
    print('Rekening: $_nomorRekening | Saldo: Rp $_saldo');
  }
}

void main() {
  var bank = RekeningBank('2514-10021', 2000000);
  bank.cekSaldo();
  bank.tarik(500000);
  bank.cekSaldo();
}
